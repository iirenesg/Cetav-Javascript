        class TaskFactory {
          constructor() {
            this.createTask = (type) => {
              let task;

              if (type === 'shopping') {
                task = new ShoppingTask();
              } else if (type === 'chores') {
                task = new ChoresTask();
              }

              return task;
            }
          }
        };

        class Task {
            constructor() {
                this.completed = false;
                // Properties
