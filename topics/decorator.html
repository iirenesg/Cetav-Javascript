Decorator Pattern

Decorator

Class definition

      
        class List {
          constructor(name, songs) {
            this.name = name;
            this.songs = songs;
          }
        }
        
        class Favorites {
          constructor(baseList) {
            this.name = 'Favorites';
            this.songs = baseList.songs;
          }

          removeSong(song) {
            console.log(`Removing this song: ${song}`);
          }
        }
        
        class Playlist {
          constructor(baseList) {
            this.name = baseList.name;
            this.songs = baseList.songs;
          }
          
          remove() {
            console.log('Removing playlist');
          }
        }

        export default List;
        export { Favorites, Playlist };

Decorator

Usage

      
        import List, { Favorites, Playlist } from './List.js';

        const list = new List('Custom List', [...]);
        const favs = new List('', [...]);

        const playlist = new Playlist(list);
        const favorites = new Favorites(favs);
        
        playlist.remove();
        favorites.removeSong({...});
